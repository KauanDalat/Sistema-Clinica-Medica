1. Criação das Tabelas  

-- Tabela Usuario 
CREATE TABLE Usuario ( 
    id_usuario INTEGER PRIMARY KEY, 
    cpf VARCHAR(14) UNIQUE NOT NULL, 
    nome VARCHAR(50) NOT NULL, 
    email VARCHAR(50) UNIQUE NOT NULL, 
    telefone VARCHAR(20) NOT NULL, 
    data_nascimento DATE, 
    genero CHAR(1) 
); 
 -- Tabela Recepcionista 
CREATE TABLE Recepcionista ( 
    id_usuario INTEGER PRIMARY KEY, 
    FOREIGN KEY (id_usuario) REFERENCES Usuario(id_usuario) 
); 
 -- Tabela Medico 
CREATE TABLE Medico ( 
    id_usuario INTEGER PRIMARY KEY, 
    FOREIGN KEY (id_usuario) REFERENCES Usuario(id_usuario) 
); 
 -- Tabela Paciente 
CREATE TABLE Paciente ( 
    id_usuario INTEGER PRIMARY KEY, 
    FOREIGN KEY (id_usuario) REFERENCES Usuario(id_usuario) 
); 
 -- Tabela Especialidade 
CREATE TABLE Especialidade ( 
    id_especialidade INTEGER PRIMARY KEY, 
    nome_especialidade VARCHAR(50) NOT NULL 
); 
 -- Tabela Medico_especialidade 
CREATE TABLE Medico_especialidade ( 
    id_medico INTEGER, 
    id_especialidade INTEGER, 
    PRIMARY KEY (id_medico, id_especialidade), 
    FOREIGN KEY (id_medico) REFERENCES Medico(id_usuario), 
    FOREIGN KEY (id_especialidade) REFERENCES Especialidade(id_especialidade) 
); 
 -- Tabela Convenio 
CREATE TABLE Convenio ( 
    id_convenio INTEGER PRIMARY KEY, 
    nome VARCHAR(50) NOT NULL, 
    cobertura VARCHAR(50) 
); 
 -- Tabela Paciente_convenio 
CREATE TABLE Paciente_convenio ( 
    id_paciente INTEGER, 
    id_convenio INTEGER, 
    numero_carteirinha VARCHAR(30), 
    PRIMARY KEY (id_paciente, id_convenio), 
    FOREIGN KEY (id_paciente) REFERENCES Paciente(id_usuario), 
    FOREIGN KEY (id_convenio) REFERENCES Convenio(id_convenio) 
); 
 -- Tabela Atendimento 
CREATE TABLE Atendimento ( 
    id_atendimento INTEGER PRIMARY KEY, 
    data_hora_atendimento DATETIME, 
    valor_consulta DECIMAL(10,2), 
    status VARCHAR(10), 
    anamnese TEXT, 
    id_recepcionista INTEGER, 
    id_medico INTEGER, 
    id_paciente INTEGER, 
    FOREIGN KEY (id_recepcionista) REFERENCES Recepcionista(id_usuario), 
    FOREIGN KEY (id_medico) REFERENCES Medico(id_usuario), 
    FOREIGN KEY (id_paciente) REFERENCES Paciente(id_usuario) 
);
